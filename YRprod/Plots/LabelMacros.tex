%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%% set of macros for easy labeling of plots
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\def\Pave#1{\put(300,250){\begin{picture}( 1200,1200)#1\end{picture}}}
\def\PaveL#1#2#3{\put(#1,#2){\makebox(0,0)[l]{\LARGE #3}}}
\def\PaveLr#1#2#3{\put(#1,#2){\makebox(0,0)[r]{\LARGE #3}}}
\def\PaveLb#1#2#3{\put(#1,#2){\makebox(0,0)[b]{\LARGE #3}}}
\def\PaveLt#1#2#3{\put(#1,#2){\makebox(0,0)[t]{\LARGE #3}}}
\endinput
